001  (ns scae-library.tokeniser
002    (:require [clojure.data.json :as json]))
003  
004  (defn- insert-token-type
005    [token-definitions]
006    (hash-map :tokens (apply merge (map #(into {} (for [entry (get token-definitions %)]
007                                                   {(first entry) {:regex (second entry)
008                                                                   :type  %}}))
009                                       (keys token-definitions)))))
010  
011  (defn- get-token-definitions
012    "Reads token definitions as a vector of token or skip hash-maps,
013    returns hashmap of the type of token and a corresponding hash-map of tokens
014    input: [{\"token\": {\"token1\": \"token-regex1\",
015                       \"token2\": \"token-regex2\"}},
016            {\"skip\": {\"skip1\": \"skip-regex1\"}},
017            {\"token\": {\"token3\": \"token-regex3\"}}]
018  
019    result: {:token {:token1 \"token-val1\",
020                     :token2 \"token-val2\",
021                     :token3 \"token-val3\"},
022             :skip  {:skip1 \"skip-val1\"}}"
023    [token-definitions-json]
024    (->> token-definitions-json
025      (apply merge-with conj)
026      (insert-token-type)))
027  
028  (defn- val->token
029    "Convert value to its corresponding token. Return nil if there is no
030    corresponding token"
031    [val token-definitions]
032    (->>
033      (map (fn
034             [token]
035             (if (re-matches (re-pattern (:regex (second token))) val)
036               {:token-key (first token)
037                :token-value    val
038                :token-type (:type (second token))}))
039            (:tokens token-definitions))
040      (filter (complement nil?))
041      (first)  ;; for an empty collection returns nil
042      ))
043  
044  (defn tokenise-code
045    [code token-vector]
046    (let [token-definitions (get-token-definitions token-vector)]
047      (as-> token-definitions tokens
048            ;;join regexes separated by | to get in order matches
049            (clojure.string/join "|" (map :regex (vals (:tokens tokens))))
050            (re-seq (re-pattern tokens) code)
051            (map #(val->token %1 token-definitions) tokens))))
