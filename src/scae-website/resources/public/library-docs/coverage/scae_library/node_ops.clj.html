024  (defn get-parsed-node-result
025    "This returns the parsed node result. If the parsed node result is a string,
026    it reads the string and returns the eval'd result. This is due to elsewhere
027    in the program, the style analyser, having to quote the parsed node result so
028    that internal hash-maps would not try to be evaluated as a function if they
029    were at the head of a list and eval were called on them"
030    [ast-node]
031    (let [parsed-node-result (:parsed-node-result ast-node)]
032      (if (string? parsed-node-result)
033        ;;If string read it, eval it, then return it
034        (eval (read-string parsed-node-result))
035        ;;else it is not a string, so can return it straight away
036        parsed-node-result)))

046  (defn node-contains-nested-token?
047    "Returns true if the node contains a token that has a value of token-value
048    with key of token-key anywhere in its children or their sub-trees"
049    [ast-entry token-key token-value]
050    (cond
051      ;;if current entry is a token, check its value and return true if match is found
052      (token? ast-entry)
053      (if (and
054            (= token-key
055               (:token-key ast-entry))
056            (= token-value
057              (:token-value ast-entry)))
058        true)
059  
060      ;;if current entry is a node, recursively check all its children
061      (node? ast-entry)
062      (let [contains-nested-value?
063            (some true?
064                  (for [child (get-parsed-node-result ast-entry)]
065                    (node-contains-nested-token? child token-key token-value)))]
066        (if contains-nested-value?
067          true))
068  
069      ;;Node does not contain nested token value, return nil
070      :else
071      nil
072      ))

074  (defn get-nth-child
075    "Get the nth child of the ast node. Uses zero indexing (0 is first position).
076    Ignores symbol table function calls and line-numbers"
077    [ast-node n]
078    (let [children (for [entry (get-parsed-node-result ast-node)
079                         ;;only compile list of nodes and tokens
080                         ;;(essentially ignore st-func-calls and line numbers)
081                         :when (or (node? entry)
082                                   (token? entry))]
083                     entry)]
084      ;;return nth element, if it doesn't exist return nil
085      (nth children n nil))
086    )

088  (defn get-nth-named-child
089    "Retrieve the nth instance of the target child. Uses zero indexing "
090    [ast-node n child-name]
091    (let [matching-children
092          (for [entry (get-parsed-node-result ast-node)
093                :when (or
094                        ;;check for specific child of type node
095                        (= child-name
096                           (:parsed-node-name entry))
097                        ;;check for specific child of type token
098                        (= child-name
099                           (:token-key entry)))]
100            entry)]
101      ;;check that there are at least n matches. If true return nth, if not return nil
102      (if (<= n (count matching-children))
103        (nth matching-children n nil)
104        nil)))

106  (defn get-in-nested-entry
107    "Gets the nested entry of the node following the path-pair. The path-pair
108    is a vector of vectors. The inner vectors have 2 values, n and child-name, which
109    are passed into the get-nth-name-child function."
110    [ast-entry path-pairs]
111    (if (empty? path-pairs)
112      ;;the current entry is our target entry
113      ast-entry
114  
115      ;;else follow the first path-pair and recursively call
116      (let [path-pair (first path-pairs)
117            n (first path-pair)
118            child-name (second path-pair)]
119        (->
120          ast-entry
121          (get-nth-named-child n child-name)
122          (get-in-nested-entry (rest path-pairs))))))

125    [ast-node child-name]
126    ;;if a single (first) instance of child exists, return true, else nil
127    (if (get-nth-named-child ast-node 0 child-name)
128      true
129      nil))
