001  (ns scae-library.style-analyser
002    (:require [scae-library.node-ops :as node-ops]
003              [clojure.string :as string]
004              [com.rpl.specter :refer :all]
005              [clojure.walk :as walk]))
006  
007  (defn quote-first-parsed-node-result
008    [ast-node]
009    ;;todo check if ordering of statement is free of side effects, check if I can get rid of specter
010    (transform [:parsed-node-result] #(str "'" (reverse (into (list) %))) ast-node))
011  
012  (defn parse-style-rules
013    [ast-node node-based-style-rules]
014    (let [current-node-suggestions
015          (for [node-key-style-vector-val-pair node-based-style-rules]
016            ;;if current node matches any style rules
017            (if (= (name (first node-key-style-vector-val-pair))
018                   (node-ops/get-node-name ast-node))
019  
020              ;;eval all style rules
021              (let [style-rule-vector (read-string (second node-key-style-vector-val-pair))]
022                (for [style-rule style-rule-vector]
023                  ;;note when putting a vector into a list it is done in reverse order
024                  ;;also note I have to insert a quoted string to prevent unintended evaluation
025                  (eval (into (list) [(quote-first-parsed-node-result ast-node) style-rule]))))
026  
027              ;;Else return an empty string
028              ""))
029          children-node-suggestions
030          (for [child (node-ops/get-node-children ast-node)
031                :when (node-ops/node? child)]
032            (parse-style-rules child node-based-style-rules))]
033  
034  
035      (filter #(not (string/blank? %))
036              (flatten
037                (concat
038                  current-node-suggestions
039                  children-node-suggestions)))))
040  
041  (defn analyse-style
042    [root-ast-node style-rules]
043    ;;define the style rules in the namespace
044    (doseq [style-function-definition (vals (:style-rules-function-definitions style-rules))]
045      (eval (read-string style-function-definition)))
046  
047    ;;parse through the AST, evaluating style rules for the nodes that have them
048    (parse-style-rules root-ast-node (:node-based-style-rules style-rules)))
