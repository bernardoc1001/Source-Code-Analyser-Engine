004  (defn code->node
005    [code]
006    (let [unevaled-code (if (string? code)
007                          (read-string code)
008                          code)]
009      (hash-map :node-name (name (first unevaled-code))
010                :node-value (eval unevaled-code))))

013  (defn count-tokens
014    [coll]
015    (->> coll
016         (tree-seq coll? seq)
017         (keep #(get-in % [:token-key]))
018         count))

023  ;;recursively check single elem of inner vec at a time to easily parse through tokens
024  (defn parse-single-option
025    [tokens option-vec]
026    ;;(println "current parse token: " (first tokens))
027    ;;(println "current parse option-vec: " option-vec)
028    (if (seq option-vec)
029      (do (if (and (list? (first option-vec)) (not-empty (first option-vec)))
030            ;; If we see a list here, thats a function call to another production rule
031            (let [inner-func-node (parse-production-rule tokens (first option-vec))]
032              (if (:parsed-node-result inner-func-node)
033                (do
034                  ;;(println "inner func node result: \n" (:parsed-node-result inner-func-node))
035                  ;;here i want to remove tokens in result from tokens, then remove head of option,
036                  ;; then continue parsing the rest of the current option vector
037                  (let [recursive-result-after-inner-func
038                        (parse-single-option (drop
039                                               (count-tokens (:parsed-node-result inner-func-node))
040                                               tokens)
041                                             (rest option-vec))]
042                    ;;(println "recursive-result-after-inner-func: " recursive-result-after-inner-func)
043                    (if recursive-result-after-inner-func
044                      (concat [inner-func-node] recursive-result-after-inner-func)))
045                  )))
046            (if (= (:token-key (first tokens)) (first option-vec))
047              (do
048                ;;(println (first tokens) " == " (first option-vec))
049                (let [recursive-call-result (parse-single-option (rest tokens) (rest option-vec))]
050                  (if recursive-call-result
051                    (concat (vector (first tokens)) recursive-call-result)
052                    (do
053                      ;;(println "recursive call result empty")
054                      nil))))
055  
056              ;;check for epsilon condition
057              (if (vector? (first option-vec))
058                ;; If we see a vector here, it's an epsilon transition
059                (do
060                  ;;(println "Epsilon condition, token:  " (first tokens) "  and option-vec: " option-vec)
061                  [])
062                (if (map? (first option-vec))
063                  ;;If we see a map here, it contains symbol table function calls in a vector.
064                  ;;Return the map and continue with the parse without evaluating the symbol table calls
065                  (do
066                    (let [continued-parsing-result (parse-single-option tokens (rest option-vec)) ;;recurisve call without consumming token
067                          ]
068                      (if continued-parsing-result
069                        (concat (vector (first option-vec)) continued-parsing-result) ;;concat the symbol table function call map with the result of the parse
070                        (do
071                          ;;(println "recursive call result empty")
072                          nil))))
073  
074  
075                  ;;else
076                  (do
077                    ;;(println (first tokens) " != " (first option-vec))
078                    nil))))))
079      [];;returning empty vec here so as not to return nil
080      ))

083  (defn parse-production-rule
084    [tokenised-code production-rule]
085    ;;(println "prod rule: " production-rule)
086    (let [filtered-tokens (filter #(= (:token-type %) :token) tokenised-code)
087          production-node (code->node production-rule)
088          all-options-parsed
089          (for [option-vec (:node-value production-node)]
090            (do
091              #_(println "node name: " (:node-name production-node))
092              (hash-map :parsed-node-name (:node-name production-node)
093                        :parsed-node-result (parse-single-option filtered-tokens option-vec))))]
094      ;;TODO remove prints
095      #_(println "CODE->Node input: \n" production-rule)
096      #_(println "CODE->NODE output: \n" production-node)
097  
098      (first (filter #(not (nil? (:parsed-node-result %))) all-options-parsed))))

100  (defn create-abstract-syntax-tree
101    [tokenised-code production-rules]
102    (eval (read-string (:scae-forward-declarations production-rules)))
103    (eval (read-string (:scae-program production-rules)))
104    (doseq [production (vals production-rules)]
105      (eval (read-string production)))
106    (parse-production-rule tokenised-code (:scae-entry-point production-rules)))
